64 void* video_routine(void* args)
65 {
66  /*TCP segment of encoded video received from the drone*/
67  static uint8_t tcp_buf[TCP_VIDEO_BUF_SIZE];
68  /*size of this segment in bytes*/
69  ssize_t pack_size = 0;
70  /*set to 1 by the decoding function if a decoded video frame is available*/
71  int got_frame = 0;
72  /*structure that will hold our latest decoded video frame*/
73  jakopter_video_frame_t decoded_frame;
74 
75  while (!video_is_stopped()) {
76  //Wait for the drone to send data on the video socket
77  if (select(sock_video+1, &vid_fd_set, NULL, NULL, &video_timeout) < 0) {
78  perror("[~][video] Cannot select()");
79  video_set_stopped();
80  break;
81  }
82 
83  if (!FD_ISSET(sock_video, &vid_fd_set)) {
84  printf("[*][video] Data reception has timed out. Ending the video thread now.\n");
85  video_set_stopped();
86  break;
87  }
88 
89  /*receive the video data from the drone. Only BASE_SIZE, since
90  TCP_SIZE may be larger on purpose.*/
91  pack_size = recv(sock_video, tcp_buf, BASE_VIDEO_BUF_SIZE, 0);
92  if (pack_size == 0) {
93  printf("[*][video] Stream ended by server. Ending the video thread.\n");
94  video_set_stopped();
95  break;
96  }
97 
98  if (pack_size < 0) {
99  perror("[~][video] Cannot recv()");
100  }
101  else {
102  //we actually got some data, send it for decoding !
103  got_frame = video_decode_packet(tcp_buf, pack_size, &decoded_frame);
104  if (got_frame < 0) {
105  fprintf(stderr, "[~][video] Cannot decoding video\n");
106  video_set_stopped();
107  break;
108  }
109  //if we have a complete decoded frame, push it onto the queue for decoding
110  else if (got_frame == 1)
111  video_queue_push_frame(&decoded_frame);
112  }
113 
114  //reset the timeout and the FDSET entry
115  video_timeout.tv_sec = VIDEO_TIMEOUT;
116  FD_ZERO(&vid_fd_set);
117  FD_SET(sock_video, &vid_fd_set);
118  nanosleep(&video_interval, NULL);
119  video_interval.tv_nsec = VIDEO_INTERVAL;
120  }
121  /*push an empty frame on the queue so the processing
122  thread knows it has to stop*/
123  video_queue_push_frame(&VIDEO_QUEUE_END);
124  pthread_join(processing_thread, NULL);
125  //there's no reason to keep stuff that's needed for our video thread once it's ended, so clean it now.
126  video_clean();
127  pthread_exit(NULL);
128 }

131 {
132  //decoded video frame that will be pulled from the queue
133  jakopter_video_frame_t frame;
134  //wait for frames to be decoded, and then process them.
135  while (!video_is_stopped()) {
136  if (video_queue_pull_frame(&frame) < 0) {
137  fprintf(stderr, "[~][Video Processing] Cannot retrieving frame\n");
138  video_set_stopped();
139  }
140  //a 0-sized frame means we're about to quit.

143  fprintf(stderr, "[~][Video Processing] Cannot processing frame\n");
144  video_set_stopped();
145  }
146  }
147  }
148  //free the resources of the processing module
149  if (frame_process.clean != NULL)
150  frame_process.clean();
151  pthread_exit(NULL);
