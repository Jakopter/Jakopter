506 static int create_cleanup_udata(lua_State* L)
507 {
508  //metatable with cleanup method for the lib
509  luaL_newmetatable(L, "jakopter.cleanup");
510  //set our cleanup method as the __gc callback
511 
512  lua_pushstring(L, "__gc");
513  lua_pushcfunction(L, jakopter_cleanup_lua);
514  lua_settable(L, -3);
515  /*use a userdata to hold our cleanup function.
516  We don't store anything in it, so its size is meaningless.*/
517  lua_pushstring(L, "jakopter.cleanup_dummy");
518  lua_newuserdata(L, 4);
519  luaL_getmetatable(L, "jakopter.cleanup");
520  lua_setmetatable(L, -2);
521  //store this dummy data in Lua's registry.
522  lua_settable(L, LUA_REGISTRYINDEX);
523  return 0;
524 }
