791 void draw_attitude_indic()

834 void rotate_point(SDL_Point* point, const SDL_Point* center, float angle)
835 {
836  //convert the angle to radians for use with C math functions.
837  double a_rad = angle * PI/180.;
838  double a_cos = cos(a_rad), a_sin = sin(a_rad);
839  //translate to origin before rotating
840  point->x -= center->x;
841  point->y -= center->y;
842  //compute the rotation
843  int newx = point->x*a_cos - point->y*a_sin;
844  int newy = point->x*a_sin + point->y*a_cos;
845  //translate back to position
846  point->x = newx + center->x;
847  point->y = newy + center->y;
848 }
